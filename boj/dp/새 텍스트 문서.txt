const fs = require('fs');
const input = fs.readFileSync('/dev/stdin').toString().trim().split("\n");

const n = +input[0];
const arr = input[1].split(' ').map(Number);
const dp = new Array(n).fill(0);
dp[0] = arr[0];

let answer = dp[0];

for (let i = 1; i < n; i++) {
  dp[i] = Math.max(dp[i - 1] + arr[i], arr[i]);
  answer = Math.max(answer, dp[i]);
}

console.log(answer);
